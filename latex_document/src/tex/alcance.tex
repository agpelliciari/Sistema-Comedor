\section{Alcance}

\subsection{Propósito del sistema}
Nuestro sistema de gestión para comedores universitarios busca modernizar la experiencia de estudiantes y personal, permitiendo a los estudiantes consultar el menú, armar, realizar y pagar pedidos desde una interfaz web, y brindando al personal herramientas para gestionar pedidos, preparación en cocina y disponibilidad de productos en tiempo real. Adicionalmente, facilita la administración del menú, stock, promociones y accesos del personal.

\subsection{Alcance funcional}
\subsubsection*{Actores principales}
\begin{itemize}
  \item \textbf{Estudiante/cliente}: se registra y autentica, gestiona su perfil, consulta menú y disponibilidad, arma, realiza y paga pedidos, consulta estado y los retira.
  \item \textbf{Personal de cocina/caja}: visualiza pedidos entrantes, cambia estados (p.ej., pedido confirmado, en preparación, completado), gestiona disponibilidad operativa.
  \item \textbf{Encargado/a del comedor}: administra menú, ingredientes, combos/paquetes, stock, promociones, y acceso del personal (roles).
\end{itemize}

\subsubsection*{Funciones incluidas}
\begin{itemize}
  \item \textbf{Gestión de usuarios}: registro con verificación de email; autenticación y recuperación de contraseña; perfil con nombre, apellido, email, foto, edad y domicilio.
  \item \textbf{Catálogo y composición de productos}:
    \begin{itemize}
      \item Ítems individuales y \emph{combos} (paquetes con precio especial).
      \item Ítems compuestos por ingredientes; dependencia de disponibilidad por ingrediente.
      \item Disponibilidad automática: si falta un ingrediente clave, los ítems/combos afectados quedan \emph{no disponibles} para los estudiantes.
    \end{itemize}
  \item \textbf{Gestión de stock}: alta/baja/ajuste de ingredientes y productos; soporte para procesos de reposición (manuales).
  \item \textbf{Promociones dinámicas}: reglas configurables (p.ej., ``con un sándwich, un café gratis'', 3x2, descuentos por día o por monto, descuentos condicionados por método de pago).
  \item \textbf{Ciclo de vida del pedido}:
    \begin{enumerate}
      \item \emph{Pedido armado/pendiente de pago} por el estudiante.
      \item \emph{Confirmado/pagado} por el estudiante.
      \item \emph{En preparación} por personal de cocina.
      \item \emph{Completado / listo para retirar}.
    \end{enumerate}
    \noindent \emph{Cancelación} permitida mientras no haya comenzado la preparación (estado en preparación o posterior).
  \item \textbf{Backoffice del encargado}: ABM de menú, ingredientes, combos, promociones y gestión de accesos del personal (asignación de roles como administrador/empleado).
  \item \textbf{Pagos}: selección de método de pago (efectivo, débito, crédito).
  \item \textbf{Notificaciones de estado}: avisos al usuario ante cambios relevantes en el estado de sus pedidos (p.ej., cuando pasa a estar en preparación o completado).
\end{itemize}

\subsection{Alcance no funcional}
\begin{itemize}
  \item \textbf{Seguridad y autenticación}: autenticación con tokens (JWT), almacenamiento seguro de credenciales, verificación de email y recuperación de contraseña; control de acceso basado en roles (cliente, empleado, administrador).
  \item \textbf{Disponibilidad}: servicio operativo en horario del comedor.
  \item \textbf{Rendimiento}: respuestas rápidas para consultas de menú bajo carga típica; actualización frecuente del estado de los pedidos en las vistas de cocina/pedidos.
  \item \textbf{Consistencia}: para operaciones críticas (estado del pedido, disponibilidad de ingredientes y aplicación de descuentos).
  \item \textbf{Compatibilidad}: API documentada (Swagger/Postman); contenedores Docker para despliegue.
\end{itemize}

\subsection{Stakeholders y usuarios}
\begin{itemize}
  \item \textbf{Estudiantes} (usuarios finales).
  \item \textbf{Personal del comedor} (cocina/caja).
  \item \textbf{Encargado/a del comedor} (administración y configuración).
  \item \textbf{Equipo de TI/desarrollo} (operación y evolución).
\end{itemize}

\subsection{MVP / Early Testable Product}
\label{subsec:mvp}
\begin{itemize}
  \item Registro, autenticación, validación de email y recuperación de contraseña.
  \item Menú con disponibilidad por stock e ingredientes.
  \item Flujo de pedido de extremo a extremo (incluyendo estados de pago, preparación y completado).
  \item Panel de cocina y backoffice mínimos.
  \item API y despliegue en Docker para pruebas.
\end{itemize}
